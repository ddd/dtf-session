def env_to_hash(env_string)
  lines = env_string.split("\n")
  key_value_pairs = lines.map { |line|
    key, value = *line.split("=", 2)
    [key.to_sym, value]
  }

  Hash[key_value_pairs]
end

def set_start_env(shell)
  env = shell.execute('printenv')
